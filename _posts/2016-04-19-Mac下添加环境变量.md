    cd ~                                            //切换到根目录
    touch .bash_profile                             //寻找或者创建文件
    open -e .bash_profile                           //打开文件
    exprot PATH=${PATH}:/Users/Library/AndroidSDK/  //添加环境变量
    source .bash_profile                            //生效配置
